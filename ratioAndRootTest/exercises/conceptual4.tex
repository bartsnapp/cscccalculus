\begin{exercise}
Suppose that $\{a_n\}_{n=1}$ is a sequence of positive terms and set $s_n = \sum_{k=1}^n a_k$.  Suppose it is known that:

\[
\lim_{n \to \infty} \frac{a_{n+1}}{a_n} = \frac{1}{9}
\]


Select all of the following that \emph{must} be true.

\begin{selectAll}
\choice[correct]{$\sum_{k=1}^{\infty} a_k$ must converge.}
\choice{$\sum_{k=1}^{\infty} a_k$ must converge to $\frac{1}{9}$.}
\choice{$\sum_{k=1}^{\infty} \frac{a_{k+1}}{a_k}$ must converge.}
\choice[correct]{$\{s_n\}$ must be bounded.}
\choice[correct]{$\{s_n\}$ is monotonic.}
\choice{$\lim_{n \to \infty} s_n$ does not exist.}
\end{selectAll}

\begin{hint}
Since $a_n \geq 0$ for all $n$, $\{s_n\}$
\begin{multipleChoice}
\choice[correct]{must be increasing.}
\choice{must be decreasing.}
\choice{could be increasing or decreasing.}
\end{multipleChoice}

The Ratio Test guarantees that $\sum_{k=1}^{\infty} a_k$:

\begin{multipleChoice}
\choice[correct]{must converge.}
\choice{must diverge.}
\choice{could converge or diverge.}
\end{multipleChoice}


\begin{question}

Since $\sum_{k=1}^{\infty} a_k$ converges, $\lim_{n \to \infty} s_n$
\begin{multipleChoice}
\choice[correct]{must exist.}
\choice{does not exist.}
\choice{could exist or not exist.}
\end{multipleChoice}

\begin{question}
Since $\lim_{n \to \infty} s_n$ exists,
\begin{multipleChoice}
\choice[correct]{$\{s_n\}$ is bounded.}
\choice{$\{s_n\}$ could be bounded or unbounded.}
\choice{$\{s_n\}$ is unbounded.}
\end{multipleChoice}

\end{question}
\end{question}

Since $\lim_{n \to \infty} \frac{a_{n+1}}{a_n} \neq 0$, the Divergence Test guarantees that

\begin{multipleChoice}
\choice{$\sum_{k=1}^{\infty} a_k$ must diverge.}
\choice[correct]{$\sum_{k=1}^{\infty} \frac{a_{k+1}}{a_k}$ must diverge.}
\end{multipleChoice}

\end{hint}

\end{exercise}
