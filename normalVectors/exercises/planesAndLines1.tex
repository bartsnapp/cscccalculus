\begin{exercise}
Which of the following represents a line that passes through the point $(2,0,3)$ and is orthogonal to the plane $x-y+3z=12$?

\begin{multipleChoice}
\choice{$\vec{r}(t) = \vector{2t+1,-1,3t+3}$}
\choice[correct]{$\vec{r}(t) = \vector{t+2,-t,3t+3}$}
\choice{$\vec{r}(t) = \vector{t,-t+4,3t2}$}
\choice{$x-y+3z = 11$}
\choice{$2x+3z=11$}
\choice{$2x-y-z=1$}
\choice{There is no such line.}
\choice{None of the above}
\end{multipleChoice}

Which of the following represents a plane that passes through the point $(2,0,3)$ and is orthogonal to the plane $x-y+3z=12$?

\begin{multipleChoice}
\choice{$\vec{r}(t) = \vector{2t+1,-1,3t+3}$}
\choice[correct]{$\vec{r}(t) = \vector{t+2,-t,3t+3}$}
\choice{$\vec{r}(t) = \vector{t,-t+4,3t2}$}
\choice{$x-y+3z = 11$}
\choice{$2x+3z=11$}
\choice[correct]{$2x-y-z=1$}
\choice{There is no such line.}
\choice{None of the above}
\end{multipleChoice}

\end{exercise}
