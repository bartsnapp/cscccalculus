\begin{exercise}
Determine the indefinite integral 

\[
\int x\sqrt{x^{2}-5} \d x
\]


\[
\int x \sqrt{x^{2}-5} \d x= \answer{ \frac{ (x^{2}-5)^{\frac{3}{2}}}{3} +C }
\]
(Use $C$ for the constant of integration)

\begin{hint}
Try a $u$-substitution first!
\end{hint}
\end{exercise}
