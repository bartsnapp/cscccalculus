\begin{problem}
  The point $(1,2,3)$ is in \textbf{cartesian coordinates.} Express
  this point in \textbf{spherical coordinates,} $(\rho,\theta,\phi)$,
  with $0\le \rho$, $0\le \phi \le \pi$, and $0\le \theta<
  2\pi$. \textbf{You do not need to simplify.}
  \begin{prompt}
    \[
    (\rho,\theta,\phi) = \left(\answer{\sqrt{14}},\answer{\arctan(2)},\answer{\arccos(3/\sqrt{14})}\right)
    \]
  \end{prompt}
\end{problem}

\begin{problem}
  Let $F:\R^2\to\R$ and $G:\R^2\to\R$ be continuous functions.
  \begin{prompt}
    Select all true statements.
  \end{prompt}
  \begin{selectAll}
    \choice{$\left(\int_{-2}^{3} \int_{-4}^5 F(x,y) \d y \d x \right)\left( \int_{-2}^{3} \int_{-4}^5 G(x,y) \d y \d x\right) = \int_{-2}^{3} \int_{-4}^5 F(x,y)\cdot G(x,y)\d y \d x$}
    \choice{If $H(x,y) = \int_0^x\int_0^y F(s,t)\d s \d t$, then $H^{(1,1)}(x,y) = F(x,y)$.}
    \choice[correct]{$\int_{-2}^{3} \int_{-4}^5 F(x,y) \d y \d x  + \int_{-2}^{3} \int_{-4}^5 G(x,y) \d y \d x = \int_{-2}^{3} \int_{-4}^5 \left(F(x,y) +G(x,y)\right)\d y \d x$}
    \choice{Let $R=\{(x,y): x^2+y^2 \le 1\}$. If $\iint_R F(x,y) \d A=0$, then $F(x,y)=0$ for all $(x,y)$ in $R$.}
    \choice[correct]{$\iint_R 10 \d A$ is $10$ times the area of $R$.}
    \choice{$\int_0^1 \int_0^x F(x,y) \d y \d x = \int_0^1\int_0^y F(x,y)\d x\d y$}
  \end{selectAll}
\end{problem}
