\begin{problem}
  The point $(-3,-2,-1)$ is in \textbf{cartesian coordinates.} Express
  this point in \textbf{spherical coordinates,} $(\rho,\theta,\phi)$,
  with $0\le \rho$, $0\le \phi \le \pi$, and $0\le \theta<
  2\pi$. \textbf{You do not need to simplify.}
  \begin{prompt}
    \[
    (\rho,\theta,\phi) = \left(\answer{\sqrt{14}},\answer{\arctan(2/3)+\pi},\answer{\arccos(-1/\sqrt{14})}\right)
    \]
  \end{prompt}
\end{problem}

\begin{problem}
  Let $F:\R^2\to\R$ be a continuous function.
  \begin{prompt}
    Select all true statements.
  \end{prompt}
  \begin{selectAll}
    \choice{$\int_{1}^{3} \int_{2}^5 F(x,y) \d y \d x + \int_{-2}^{1} \int_{-4}^2 F(x,y) \d y \d x = \int_{-2}^{3} \int_{-4}^5 F(x,y) \d y \d x$}
    \choice[correct]{$\int_0^{2\pi}\int_{0}^1 r \d r \d \theta<\int_{-1}^1\int_{-1}^1 \d y \d x$}
    \choice[correct]{The integral  $\int_{-1}^1\int_{-\sqrt{1-x^2}}^{\sqrt{1-x^2}} \d y \d x$ is the area of the unit disk.}
    \choice{$\int_0^1\int_0^x F(x,y) \d y \d x \le \int_0^1\int_0^1 F(x,y) \d y \d x$}
    \choice{The surface area of $z = x^2+y^2$ over some region $R$ is given by $\iint_R (x^2+y^2) \d A$.}
    \choice[correct]{If $H(x,y) = \int_0^x\int_0^y F(s,t)\d t \d s$, then $H^{(1,1)}(x,y) = F(x,y)$.}
  \end{selectAll}
\end{problem}

\begin{problem}
  \textbf{Write} $\d V$ for the following coordinate systems:
  \begin{itemize}
  \item[Cartesian:] $\d V = \answer{1}\d x\d y\d z$
  \item[Cylindrical:] $\d V = \answer{r} \d z \d r \d \theta$
  \item[Spherical:] $\d V = \answer{\rho^2\sin(\phi)} \d \rho \d \phi \d \theta$
  \end{itemize}
\end{problem}

\begin{problem}
  Let:
  \begin{align*}
    R &= \{(x,y): x^2 +y^2 \le 1~\text{with}~x\ge0, y\ge0\}\\
    T &= \{(x,y): x+y\le 1~\text{with}~x\ge0, y\ge0\}
  \end{align*}
  Additionally, let $F(x,y) = \sqrt{1-x^2-y^2}$ and $G(x,y) = 1-x-y$.
  Which quantity is largest?
  \begin{multipleChoice}
    \pdfOnly{\begin{multicols}{2}}
      \choice[correct]{$\iint_R  \sqrt{1+F^{(1,0)}(x,y)^2+F^{(0,1)}(x,y)^2} \d A$}
      \choice{$\iint_T \sqrt{1+G^{(1,0)}(x,y)^2+G^{(0,1)}(x,y)^2} \d A$}
      \pdfOnly{\end{multicols}}
  \end{multipleChoice}
  \end{problem}
