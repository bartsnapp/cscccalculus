\begin{exercise}
Suppose that $\vec{u}$ and $\vec{v}$ are nonzero three dimensional vectors and let $\dotp$ denote the vector dot product and $\cdot$ denote scalar multiplication.  Select all of the following that must be true.  Try to think about each statement both algebraically and geometrically.

\begin{selectAll}

\choice{$\scal_\vec{u}(\vec{u}) = \vec{u}$.}
\choice[correct]{If $\vec{u}$ and $\vec{v}$ are parallel, then $\proj_\vec{v}(\vec{u}) = \vec{u}$.}
\choice[correct]{If $\vec{v}$ and $\vec{w}$ are parallel, then $\proj_\vec{v}(\vec{u}) = \proj_\vec{w}(\vec{u})$.}
\choice{If $\proj_\vec{v}(\vec{u}) \dotp \vec{v} = \vec{0}$, then $\vec{u}$ and $\vec{v}$ are parallel.}
\choice{$\vec{u}+\left|\proj_\vec{v}(\vec{u})\right|$ is defined.}
\choice[correct]{If $\uvec{w}$ is a unit vector, then $\vec{u} \dotp \uvec{w} = \scal_\uvec{w}(\vec{u})$.}
\end{selectAll}
\end{exercise}
