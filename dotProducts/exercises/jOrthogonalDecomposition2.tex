\begin{exercise}
Suppose that $\vec{u} = \vector{1,3}$ and $\vec{v} = \vector{-2,4}$.  Find a vector $\vec{P}$ parallel to $\vec{v}$ and a vector $\vec{N}$ orthogonal to $\vec{v}$ so $\vec{u} = \vec{P} +\vec{N}$.

\begin{align*}
\vec{P} = \vector{\answer{-1},\answer{2}} \\
\vec{N}= \vector{\answer{2},\answer{1}} \\
\end{align*} 

\begin{hint}
  Consider the vectors below:
  \begin{image}
    \begin{tikzpicture}
        \begin{axis}[
            unit vector ratio*=1 1 1,
            axis lines=none
          ]
          \addplot[very thick,penColor,->] plot coordinates {(0,0) (1,3)};
          \addplot[very thick,penColor2,->] plot coordinates {(0,0) (-2,4)};
          \addplot[very thick,penColor5,->] plot coordinates {(-1,2) (1,3)};
            \addplot[very thick,penColor5,->] plot coordinates {(0,0)(-1,2)};
          \node[left] at (axis cs:1, 1.2) [penColor] {$\vec{u}$};
          \node[above] at (axis cs:-2, 3) [penColor2] {$\vec{v}$};
                    \node[left] at (axis cs:-.6, 1) [penColor5] {$\vec{P}$};
          \node[above] at (axis cs:-.3, 2.5) [penColor5] {$\vec{N}$};
        \end{axis}
    \end{tikzpicture}
  \end{image}
 \end{hint}
 
 From the picture, we should proceed by:
 
 \begin{multipleChoice}
 \choice{Find $\vec{N}$ by computing $\vec{u} \cross \vec{v}$, then find $\vec{P}$ from $\vec{P} = \vec{u}-\vec{N}$.}
 \choice[correct]{Find $\vec{P}$ by computing $\vec{u} \cross \vec{v}$, then find $\vec{N}$ from $\vec{N} = \vec{u}-\vec{P}$.}
 \end{multipleChoice}
 
 \begin{exercise}
 To check that the vector $\vec{N}$ we found is orthogonal to $\vec{v}$, we should:
 
 \begin{multipleChoice}
 \choice{Verify $\vec{u} \dotp \vec{v} = 0$}
 \choice{Verify $\vec{u} \dotp \vec{N} = 0$}
  \choice[correct]{Verify $\vec{v} \dotp \vec{N} = 0$}
   \choice{Verify $\vec{u} \cross \vec{v} = 0$}
 \choice{Verify $\vec{u} \cross \vec{N} = 0$}
  \choice{Verify $\vec{v} \cross \vec{N} = 0$}
 \end{multipleChoice}
 
 Computing this, we find  $\vec{v} \dotp \vec{N} = \answer{0}$, so $\vec{v}$ and $\vec{N}$:
 
 \begin{multipleChoice}
 \choice[correct]{are orthogonal.}
 \choice{are not orthogonal.}
 \end{multipleChoice}
 \end{exercise}
\end{exercise}
