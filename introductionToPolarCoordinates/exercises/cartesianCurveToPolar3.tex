\begin{exercise}

Express the line $y= 4$ using polar coordinates:

\[
r= \answer{4\csc(\theta)}
\]

Express the line $x=5$ using polar coordinates:

\[
r= \answer{5\sec(\theta)}
\]

Express the line $y=x$ using polar coordinates:

\[
\theta= \answer{\frac{\pi}{4}}
\]

\end{exercise}
