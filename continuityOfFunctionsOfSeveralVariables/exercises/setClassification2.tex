\begin{exercise}
  Consider the function $F(x,y) = \frac{\ln(2x-y+1)}{x^2+y^2}$.
  
  Which of the following sets is the domain of $F(x,y)$?
  \begin{multipleChoice}
  \choice{$\R$}
  \choice{$\R^2$}
  \choice{$\left\{(x,y) \in \R^2 \big| 2x-y+1 >0 \right\}$}
    \choice{$\left\{(x,y) \in \R^2 \big| 2x-y+1 >0 \right\}$}
  \choice[correct]{$\left\{(x,y) \in \R^2 \big| 2x-y+1 >0, (x,y) \neq (0,0) \right\}$}
    \choice{$\left\{(x,y) \in \R^2 \big| 2x-y+1 \geq 0 \right\}$}
  \choice{$\left\{(x,y) \in \R^2 \big| 2x-y+1 \geq 0, (x,y) \neq (0,0) \right\}$}
  \end{multipleChoice}
 
 Which of the following are interior points of the domain of $F(x,y)$?
 \begin{selectAll}
 \choice{$(0,0)$}
 \choice{$(0,1)$}
 \choice[correct]{$(1,1)$}
 \choice[correct]{$(4,8)$}
  \choice{$(8,4)$}
 \end{selectAll} 

 Which of the following are boundary points of the domain of $F(x,y)$?
 \begin{selectAll}
 \choice[correct]{$(0,0)$}
 \choice[correct]{$(0,1)$}
 \choice{$(1,1)$}
 \choice{$(4,8)$}
\choice{$(8,4)$}
 \end{selectAll} 

Select the correct responses.

The domain of $F(x,y)$ is \wordChoice{\choice[correct]{open}\choice{closed}\choice{neither open nor closed}} and \wordChoice{\choice{bounded}\choice[correct]{unbounded}} .
  \end{exercise}
