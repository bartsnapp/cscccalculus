\begin{exercise}
  Consider $F(x,y) = \frac{x^3 y - y^3 x}{x^4+y^4}$. Let $C$ be the
  curve described by
  \[
  \pt{p}(t) = \point{2t,t}
  \]
  for $t> 0$. Is $C$ a level curve of $F$?
  \begin{multipleChoice}
    \choice[correct]{Yes}
    \choice{No}
  \end{multipleChoice}
  \begin{exercise}
    Which level curve is it?
    \[
    F(x,y) = \answer{6/17}
    \]
  \end{exercise}
  \begin{exercise}
    Compute the limit below, if it does not exist, write $\verb|DNE|$.
    \[
    \lim_{\point{x,y}\to \point{0,0}} F(x,y) = \answer{DNE}
    \]
  \end{exercise}
  \begin{exercise}
    Is the following function continuous on $\R^2$?
    \[
    G(x,y) =
    \begin{cases}
      F(x,y) &\text{if $(x,y)\ne (0,0)$},\\
      0 &\text{if $(x,y) = (0,0)$}.
    \end{cases}
    \]
    \begin{multipleChoice}
      \choice{Yes}
      \choice[correct]{No}
    \end{multipleChoice}
  \end{exercise}
\end{exercise}
