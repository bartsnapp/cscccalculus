\outcome{Compute integrals involving powers of sine and cosine,  Recognize the patterns that appear in trigonometric integrals,
  and use appropriate substitutions to compute them.}


\begin{document}
\begin{exercise}
  Consider the following integral:
  \[
  \int \sec^{90}(x) \cot^{59}(x)  \d x
  \]
  After making the substitution, $u = \sin(x)$ we find
  \begin{hint}
    First express secant and cotangent in terms of cosine and sine.
  \end{hint}
  \begin{multipleChoice}
    \choice[correct]{$\int u^{-59}(1-u^2)^{-16} \d u$}
    \choice{$\int u^{90}(1-u^2)^{29} \d u$}
    \choice{$\int u^{29}(1-u^2)^{90} \d u$}
    \choice{$\int u^{-18}(1-u^2)^{-59} \d u$}
  \end{multipleChoice}
\end{exercise}
