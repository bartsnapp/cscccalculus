\begin{exercise}
Consider the following integral:
  \[
  \int \sec^{27}(x)\csc^{83}(x) \d x
  \]
  After making the substitution, $u = \cos(x)$ we find
  \begin{hint}
    First express secant and cosecant in terms of cosine and sine.
  \end{hint}\begin{multipleChoice}
    \choice{$\int -(1-u^2)^{13} u^{83} \d u$}
    \choice{$\int -(1-u^2)^{27} u^{83} \d u$}
    \choice[correct]{$\int -(1-u^2)^{-42} u^{-27} \d u$}
    \choice{$\int -(1-u^2)^{-27} u^{-42} \d u$}
  \end{multipleChoice}
\end{exercise}
