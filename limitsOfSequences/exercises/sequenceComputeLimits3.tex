\begin{exercise}

Consider$\{a_n \}_{n=1}$ where $a_n =\left(\frac{1}{n}\right)^{\ln(n)}$.  Then:
\[
\lim_{n \to \infty} a_n = \answer{0}
\]

\begin{hint}
If we think of $n \to \infty$, do we wind up with an indeterminate form?
\end{hint}

\end{exercise}
