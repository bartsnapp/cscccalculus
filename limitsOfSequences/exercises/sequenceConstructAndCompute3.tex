\begin{exercise}

Consider the sequences $\{a_n \}_{n=1}$ and $\{b_n \}_{n=1}$, where $a_n =n^7+5^n$ and $b_n = 5^n$.  Then:
\[
\lim_{n \to \infty} \frac{a_n}{b_n} = \answer{1}
\]

\end{exercise}
