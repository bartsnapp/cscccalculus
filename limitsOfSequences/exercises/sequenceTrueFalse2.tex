\begin{exercise}

Select all of the following statements that are \emph{true}

\begin{selectAll}
\choice[correct]{A sequence that has a limit must be bounded.}
\choice{If a sequence has a limit, it must be bounded and monotonic.}
\choice[correct]{A sequence that is increasing is bounded below by its first term.}
\choice[correct]{If $\{a_n\}$ and $\{b_n\}$ are both bounded, then $\{a_n+b_n\}$ is bounded.}
\choice[correct]{A bounded, monotonic sequence must have a limit.}
\choice{A monotonic sequence must have a (finite) limit.}
\end{selectAll}

\end{exercise}
