\begin{exercise}
Suppose that $\vec u $ and $\vec v $ are nonzero three dimensional
vectors and $\theta$ denotes the angle between $\vec u$ and $\vec
v$. Select all the statements that \emph{must} be true.

\begin{selectAll}
\choice{$|3 \vec u + \vec v | = 3 |\vec u | + |\vec v|$}
\choice[correct]{$\proj_{\vec u} \left(\vec u \cross \vec v\right)  =\vec{0}$}
\choice[correct]{If $\proj_{\vec v} \vec u =\vec u$, then $\vec u$ and $\vec v$ are parallel.}
\choice{$ (\vec u \cross \vec v) \cross \vec v = \vec 0$}
\choice[correct]{If $\vec u = \vec v $,  then $|\vec u | = | \vec v |$.}
\choice[correct]{$ \vec u \dotp \vec v = \vec v \dotp \vec u$.}
\end{selectAll}

\begin{hint}
If you are unsure of a statment test it on easy to use vectors like $\vector{0,1} =\vec u , \vector{1,0} = \vec v$.
\end{hint}

\begin{hint}
Remember that $\proj_{\vec v} \vec u$ is in the direction of $\vec v$.
\end{hint}

\begin{hint}
Remember that $\vec u \cross \vec v$ is orthogonal to both $\vec v$ and $\vec u$.
\end{hint}
