\begin{exercise}
Select all of the following statements that \emph{must} be true.

\begin{selectAll}
\choice [correct] {If $a >0$ is a scalar, then $| a \vec u | = a |\vec u |$}
\choice [correct] {Given any nonzero vector $\vec u$, the vector $\frac{ \vec u } {|\vec u|} $ has magnitude 1.}
\choice [correct] {Suppose that $\vec u$ and $\vec v$ are nonzero three dimensional vectors and $\vec u \cdot \vec v =0$. Then, $\vec u$ and $\vec v$ are orthogonal.}
\choice[correct]{If $\vec v$ is a unit vector and $\vec u$ is any nonzero vector, $scal_{\vec v} \vec u = \vec u \cdot \vec v$. }
\choice[correct]{If $\vec u$ and $\vec v$ are othogonal unit vectors, then $\vec u \times \vec v$ is a unit vector.}
\end{selectAll}

\begin{hint}
Remember that $proj_{\vec v} \vec u$ is in the direction of $\vec v$.
\end{hint}

\begin{hint}
Since $\vec u$ and $\vec v$ are parallel then there is some $c$ such that $\vec u = c \vec v$.
\end{hint}

\begin{hint}
Remember that $\vec u \times \vec v$ is orthogonal to both $\vec v$ and $\vec u$.
\end{hint}
\end{exercise}
